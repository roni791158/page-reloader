Common Commands:

# Add URL
page-reloader add-url "https://example.com"

# Set custom interval
page-reloader set-url-interval "https://example.com" 600

# Start/Stop service
page-reloader start
page-reloader stop
page-reloader restart

# Check status
page-reloader status
page-reloader list-urls

# View logs
page-reloader logs

# Set timing presets
page-reloader set-preset fast        # 15 seconds
page-reloader set-preset normal      # 30 seconds
page-reloader set-preset slow        # 60 seconds
page-reloader set-preset very-slow   # 300 seconds (5 minutes)
